from os import name

# Create a price range column to simplify plotting later on
price_val = games_paid["price"]
games_paid['price range'] = "Not Assigned"

# Create and assign the price ranges based on pricing criteria
games_paid.loc[price_val < 10, 'price range'] = "Less than $9.99"
games_paid.loc[(price_val <20) & (price_val >= 10), 'price range'] = "$10 - $19.99"
games_paid.loc[(price_val < 40) & (price_val >= 20), 'price range'] = "$20 - $39.99"
games_paid.loc[price_val >= 40, 'price range'] = "$40+"

##############################################################################################################################################################################

''' Created a function to get the main genre (first index of genre as categorized by Steam) for each game. For the purposes of our research, we will
not count "Indie" as a game genre, but as a way to categorize the production of certain games (Indie, AA, AAA, etc.)
Therefore, if a game's main genre is Indie, we skip over it and use the next genre in the list.'''

def get_main_genre(game):
  if game is nan:
    return

  genres = game.split(',')

  if (str(genres[0]).upper == "INDIE") & (len(genres) > 1): # Some Indie observations will make it into the genre list if it is their only genre listed
    main_g = genres[1]
  else:
    main_g = genres[0]

  return str.strip(main_g) # Returns the genre, stripped of spaces in the front and back of the string

# apply the function to create the new column "main genre"
games_paid['main genre'] = games_paid['genres'].apply(get_main_genre)
genre_list = games_paid['main genre'].unique() # Lists all the genres gathered form the dataset

##############################################################################################################################################################################

''' Created a filtered DataFrame containing games that are not categorized as Indie. This is to account for the fact that
many indie games are typically priced under $20, which could visually skew plots. I seperated them for easier comparison, just in case I need them'''

non_indie = games_paid.loc[~games_paid['genres'].str.contains("Indie", na=False), ["name", "price", "main genre", "price range"]].reset_index()
indie = games_paid.loc[games_paid['genres'].str.contains("Indie", na=False), ["name", "price", "main genre", "genres", "price range"]].reset_index()
